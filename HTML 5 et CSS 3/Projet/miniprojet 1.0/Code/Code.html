Module de saisie de la grille a lettre

				

	char remplissage(char T[taille][taille])		
	{	
		for(int i=0;i< taille;i++)	
		{
			for(int j=0;j< taille;j++)
			{
				printf("donner la valeur de la case [%d][%d] : ",i+1,j+1);
				scanf("%s",&T[i][j]);
				system("cls");
			}
		}
		return(T[taille][taille]);
	}

	char affichage(char T[taille][taille])	
	{
		printf("     1   2   3   4\n");
				printf("   -----------------\n");

		{
			printf(" %d ",i+1);

			}
			printf("|\n   -----------------\n");	
		}

	}
